二、Service服务参数说明
1、pcdnservice.servcice
-A参数：用于随机自动生成sn号，默认false(可选)，随机sn号必选
--vendor_prefix: 用于指定生成sn号前缀（可选）
--sn_length：用于指定设备号前缀，默认为16最短不能低于10位（可选）
--isoutside：用于指定国内外插件使用,默认为false代表国内，true代表国外（可选），国外必选
--remote：用于指定后台API地址，联系技术获取（必选）
[Unit]
Description=pcdnnc
After=network.target

[Service]
ExecStart=/usr/bin/pcdnclient -A=xxx --vendor_prefix=xxx --sn_length=xx -isoutside=xx --remote=url
Restart= on-failure
RestartSec= 5
User=root
Group=root
KillMode=process
StandardOutput=file:/var/log/pcdn.log
StandardError=file:/var/log/pcdnerr.log
[Install]
WantedBy=multi-user.target
2、goupdate.servcice
--zone：用于指定国内外.inside国内，outside国外(必选)
--OSARCH：用于指定操作系统无需修改（必选）
[Unit]
Description=goupdate
After=network.target

[Service]
ExecStart=/usr/bin/goupdate --zone=xxxx --OSARCH=armlinux #确认好国内外修改

Group=root
KillMode=process
StandardOutput=file:/var/log/pcdn.log

[Install]
WantedBy=multi-user.target
3、固定uid.sh说明
#!/bin/bash
# sn --prefix="paopaoarm-" >> /var/log/start.log 2>&1
# file_path="/etc/pcdnuuid"
# cat "$file_path" | qrencode -m 2 -t ansi > /etc/issue
sn=`cat /proc/cmdline | grep -o 'snum=[^ ]*' |cut -d= -f2` #固定sn下，根据自身需要固定sn位置，修改变量来取sn号，目前使用亿驰演示
if [ -z "$sn" ]; then
    echo "sn is empty" >> /var/log/start.log
    exit 1
else
    deviceID=$sn
    echo $deviceID > /etc/pcdnuuid
    current_time=$(date +%Y-%m-%d\ %H:%M:%S.%3N)
    cat /etc/pcdnuuid | qrencode -m 2 -t ansi > /etc/issue
    echo $current_time >> /var/log/start.log
fi
